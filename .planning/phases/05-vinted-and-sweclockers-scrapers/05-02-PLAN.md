```javascript
function classifyCategory(keyword) {
  const kw = keyword.toLowerCase();
  if (/gpu|rtx|gtx|rx\s?\d|grafik|geforce|radeon/.test(kw)) return 'gpu';
  if (/cpu|ryzen|intel|core\s?i\d|processor|xeon/.test(kw)) return 'cpu';
  if (/ram|ddr\d|minne/.test(kw)) return 'ram';
  if (/ssd|hdd|nvme|m\.2|storage|lagring/.test(kw)) return 'storage';
  return 'unknown';
}
```

Copy `classifyCategory` from blocket.js verbatim. No `sleep` helper needed (single RSS fetch, no per-keyword loop).

**scrape(keywords) method:**
- `keywords` parameter is accepted (satisfies IScraper interface) but not used for individual HTTP requests — one RSS fetch covers all categories
- Fetch `https://www.sweclockers.com/feeds/marknad` with headers: User-Agent Chrome/120, Accept `application/rss+xml, application/xml, text/xml, */*`, Accept-Language sv-SE
- On non-ok response: `logger.warn({ status: res.status }, 'Sweclockers: RSS fetch non-200')` → return []
- On network error: `logger.error({ err: err.message }, 'Sweclockers: RSS fetch error')` → return []
- On success: `const xml = await res.text(); return this._parse(xml)`

**_parse(xml) method:**
- `const $ = cheerio.load(xml, { xmlMode: true })`
- Initialize: `const listings = []; const seenIds = new Set();`
- `$('item').each((_, el) => { ... })` — iterate all RSS items
- Get raw title: `const rawTitle = $('title', el).text().trim()`
- Filter: `if (!rawTitle.startsWith('[Säljes]')) return;` — skip [Köpes], [Bytes], [Skänkes]
- Strip prefix: `const title = rawTitle.replace(/^\[Säljes\]\s*/, '').substring(0, 250)`
- Get link: `let link = $('link', el).text().trim()` — if empty, fallback to `$('guid', el).text().trim()` (guid is confirmed permaLink in Sweclockers RSS)
- Skip if no link after fallback
- Extract ID: `const idMatch = link.match(/\/marknad\/(\d+)/)`; skip item if no match
- `const listingId = idMatch[1]`; skip if `seenIds.has(listingId)`; add to seenIds
- Description for price: `const description = $('description', el).text()`
- `const price_sek = parsePrice(description) ?? 0`
- Category: `const category = classifyCategory(title)` — classify on title, NOT keyword
- Push: `{ id: 'sweclockers:' + listingId, marketplace: 'sweclockers', title, price_sek, url: link, category, listingType: 'buy_now', auctionEndsAt: null }`
- After loop: `logger.info({ count: listings.length }, 'Sweclockers scrape complete')`
- Return listings

Export: `module.exports = { SweclockersScraper };`

**Critical:** Do NOT call `classifyCategory(keyword)` — for Sweclockers there is no per-keyword loop; call `classifyCategory(title)` on each listing title. This is the key difference from Blocket/Tradera.

Exact structure:
```
'use strict';
const { SweclockersScraper } = require('./sweclockers');
const { filterAndMarkSeen, countSeen } = require('../db/seenListings');
const logger = require('../utils/logger');

const scraper = new SweclockersScraper();

async function runSweclockers(keywords) {
  const isSeedMode = process.env.SEED_MODE === 'true' || countSeen('sweclockers') === 0;
  if (isSeedMode) {
    logger.info(
      { reason: process.env.SEED_MODE === 'true' ? 'SEED_MODE env' : 'empty seen_listings' },
      'Sweclockers: seed mode active'
    );
  }
  const listings = await scraper.scrape(keywords);
  if (isSeedMode) {
    filterAndMarkSeen(listings);
    logger.info({ count: listings.length }, 'Sweclockers seed mode: all listings marked seen, no alerts emitted');
    return [];
  }
  const novel = filterAndMarkSeen(listings);
  logger.info({ found: listings.length, novel: novel.length }, 'Sweclockers run complete');
  return novel;
}

module.exports = { runSweclockers };
```
