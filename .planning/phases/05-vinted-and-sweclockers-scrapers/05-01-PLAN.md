  <action>
Create `src/scrapers/vinted.js` using CommonJS (`'use strict'`). Use `require('undici').fetch` (same as blocket.js). Copy `classifyCategory` and `sleep` from blocket.js verbatim.

**Session cookie bootstrap (_fetchSessionCookie):**
- GET `https://www.vinted.se` with browser-realistic headers (User-Agent Chrome/120, Accept text/html, Accept-Language sv-SE)
- Extract cookie via `res.headers.get('set-cookie')` first; if null, try `res.headers.getSetCookie()` (undici extension that returns an array — take element [0])
- Match `/_vinted_\w+_session=([^;]+)/` against the cookie string
- On any error: `logger.warn` and return null (API call proceeds without cookie)

**Per-keyword API fetch (_fetchKeyword(keyword, sessionCookie)):**
- URL: `https://www.vinted.se/api/v2/catalog/items?${new URLSearchParams({ search_text: keyword, per_page: '96', order: 'newest_first' })}`
- Headers: User-Agent Chrome/120, Accept application/json, Accept-Language sv-SE; add `Cookie: _vinted_fr_session=${sessionCookie}` only when sessionCookie is non-null
- Status 429 or >= 500: `logger.warn({ status, url }, 'Vinted: rate limited or server error — skipping keyword')` → return []
- Status 403: check `res.headers.get('cf-ray') || res.headers.get('cf-mitigated')` to detect Cloudflare; `logger.warn({ status, cloudflare: !!cfHeader, url }, 'Vinted: blocked — skipping keyword')` → return []
- Any other non-ok: `logger.warn({ status, url }, 'Vinted: non-200 response')` → return []
- On network error: `logger.error({ err: err.message, url }, 'Vinted: fetch error')` → return []
- Parse: `const data = await res.json(); return this._parse(data.items || [], keyword)`

**Item normalization (_parse(items, keyword)):**
- Log first raw item in development: `logger.info({ rawItem: items[0] }, 'Vinted raw item sample')` — only when items.length > 0
- For each item: `id = 'vinted:' + item.id`, `title = String(item.title || '').substring(0, 250)`, `price_sek = Math.round(parseFloat(item.price || '0'))`, `url = item.url || 'https://www.vinted.se/items/' + item.id`, `category = classifyCategory(keyword)`, `listingType = 'buy_now'`, `auctionEndsAt = null`
- Skip items where `title` is empty after trim
- Return deduplicated array (Set on `id`)

**scrape(keywords) method:**
- Call `_fetchSessionCookie()` once at the start, store result
- Loop over keywords: call `_fetchKeyword(keyword, sessionCookie)`, push results to allListings
- Add jitter delay between keywords (skip before first): `await sleep(1000 + Math.floor(Math.random() * 2000))` (1–3 s)
- After loop: deduplicate allListings by id using a Set
- Log: `logger.info({ count: deduped.length }, 'Vinted scrape complete')`
- Return deduped array

Export: `module.exports = { VintedScraper };`

```
'use strict';
const { VintedScraper } = require('./vinted');
const { filterAndMarkSeen, countSeen } = require('../db/seenListings');
const logger = require('../utils/logger');

const scraper = new VintedScraper();

async function runVinted(keywords) {
  const isSeedMode = process.env.SEED_MODE === 'true' || countSeen('vinted') === 0;
  // log seed mode if active
  const listings = await scraper.scrape(keywords);
  if (isSeedMode) {
    filterAndMarkSeen(listings);
    logger.info({ count: listings.length }, 'Vinted seed mode: all listings marked seen, no alerts emitted');
    return [];
  }
  const novel = filterAndMarkSeen(listings);
  logger.info({ found: listings.length, novel: novel.length }, 'Vinted run complete');
  return novel;
}

module.exports = { runVinted };
```
